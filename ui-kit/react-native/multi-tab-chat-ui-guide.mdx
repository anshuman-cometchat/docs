**Quick Reference for AI Agents & Developers**

```tsx
import { createBottomTabNavigator } from "@react-navigation/bottom-tabs";
import { CometChatConversations, CometChatUsers, CometChatGroups, CometChatCallLogs } from "@cometchat/chat-uikit-react-native";

const Tab = createBottomTabNavigator();

// Tab screens with CometChat components
<Tab.Navigator>
  <Tab.Screen name="Chats" component={ConversationsScreen} />
  <Tab.Screen name="Users" component={UsersScreen} />
  <Tab.Screen name="Groups" component={GroupsScreen} />
  <Tab.Screen name="Calls" component={CallLogsScreen} />
</Tab.Navigator>

// Navigate to messages on item press
<CometChatConversations onItemPress={(conversation) => navigation.navigate("Messages", { conversation })} />
<CometChatUsers onItemPress={(user) => navigation.navigate("Messages", { user })} />
<CometChatGroups onItemPress={(group) => navigation.navigate("Messages", { group })} />
```

- **Conversations** → [CometChatConversations](/ui-kit/react-native/conversations)
- **Users** → [CometChatUsers](/ui-kit/react-native/users)
- **Groups** → [CometChatGroups](/ui-kit/react-native/groups)
- **Call Logs** → [CometChatCallLogs](/ui-kit/react-native/call-logs)
</Info>

## Overview

This guide shows you how to build a tabbed chat layout using React Navigation's bottom tabs with CometChat UI Kit components. You'll create a multi-tab interface with Conversations, Users, Groups, and Call Logs tabs, and navigate to the Messages screen when an item is selected.

<Note>
The React Native UI Kit v5 does not include a built-in tab component. Instead, use React Navigation or any other navigation library of your choice to create tabbed layouts with CometChat components.
</Note>

---

## Prerequisites

Before starting, ensure you have:

1. CometChat UI Kit installed and initialized ([Getting Started](/ui-kit/react-native/getting-started))
2. A logged-in user via `CometChatUIKit.login()` (see [Methods](/ui-kit/react-native/methods))
3. React Navigation installed in your project

<Tabs>

npm install react-native-screens react-native-safe-area-context react-native-gesture-handler

## Implementation

<Steps>
  <Step title="Create the Tab Screens">
    Create individual screen components for each tab. Each screen renders a CometChat component and handles navigation to the Messages screen.

    <Tabs>
    <Tab title="TypeScript">
    ```tsx
    // screens/ConversationsScreen.tsx
    import React from "react";
    import { View } from "react-native";
    import { CometChat } from "@cometchat/chat-sdk-react-native";
    import { 
      CometChatConversations, 
      CometChatUiKitConstants 
    } from "@cometchat/chat-uikit-react-native";
    import { NativeStackNavigationProp } from "@react-navigation/native-stack";

    type Props = {
      navigation: NativeStackNavigationProp<any>;
    };

    export function ConversationsScreen({ navigation }: Props) {
      const handleItemPress = (conversation: CometChat.Conversation) => {
        const conversationType = conversation.getConversationType();
        
        if (conversationType === CometChatUiKitConstants.ConversationTypeConstants.user) {
          const user = conversation.getConversationWith() as CometChat.User;
          navigation.navigate("Messages", { user });
        } else {
          const group = conversation.getConversationWith() as CometChat.Group;
          navigation.navigate("Messages", { group });
        }
      };

      return (
        <View style={{ flex: 1 }}>
          <CometChatConversations
            onItemPress={handleItemPress}
            hideHeader={true}
          />
        </View>
      );
    }
    ```
    </Tab>

  <Step title="Create the Messages Screen">
    Create a Messages screen that displays the chat interface using `CometChatMessageList`, `CometChatMessageHeader`, and `CometChatMessageComposer`.

    <Tabs>
    <Tab title="TypeScript">
    ```tsx
    // screens/MessagesScreen.tsx
    import React from "react";
    import { View } from "react-native";
    import { CometChat } from "@cometchat/chat-sdk-react-native";
    import {
      CometChatMessageHeader,
      CometChatMessageList,
      CometChatMessageComposer,
    } from "@cometchat/chat-uikit-react-native";
    import { RouteProp } from "@react-navigation/native";
    import { NativeStackNavigationProp } from "@react-navigation/native-stack";

    type MessagesRouteParams = {
      Messages: {
        user?: CometChat.User;
        group?: CometChat.Group;
        conversation?: CometChat.Conversation;
      };
    };

    type Props = {
      route: RouteProp<MessagesRouteParams, "Messages">;
      navigation: NativeStackNavigationProp<any>;
    };

    export function MessagesScreen({ route, navigation }: Props) {
      const { user, group } = route.params;

      return (
        <View style={{ flex: 1 }}>
          {/* Message Header */}
          <CometChatMessageHeader
            user={user}
            group={group}
            onBack={() => navigation.goBack()}
            showBackButton
          />

          {/* Message List */}
          <CometChatMessageList user={user} group={group} />

          {/* Message Composer */}
          <CometChatMessageComposer user={user} group={group} />
        </View>
      );
    }
    ```

  <Step title="Set Up Navigation">
    Configure React Navigation with bottom tabs and a stack navigator for the Messages screen.

    <Tabs>
    <Tab title="TypeScript">
    ```tsx
    // App.tsx
    import React from "react";
    import { NavigationContainer } from "@react-navigation/native";
    import { createNativeStackNavigator } from "@react-navigation/native-stack";
    import { createBottomTabNavigator } from "@react-navigation/bottom-tabs";
    import { SafeAreaProvider } from "react-native-safe-area-context";
    import { GestureHandlerRootView } from "react-native-gesture-handler";
    import Icon from "react-native-vector-icons/Ionicons";
    import {
      CometChatThemeProvider,
      CometChatI18nProvider,
    } from "@cometchat/chat-uikit-react-native";

    // Import screens
    import { ConversationsScreen } from "./screens/ConversationsScreen";
    import { UsersScreen } from "./screens/UsersScreen";
    import { GroupsScreen } from "./screens/GroupsScreen";
    import { CallLogsScreen } from "./screens/CallLogsScreen";
    import { MessagesScreen } from "./screens/MessagesScreen";

    const Tab = createBottomTabNavigator();
    const Stack = createNativeStackNavigator();

    // Tab Navigator Component
    function TabNavigator() {
      return (
        <Tab.Navigator
          screenOptions={{
            tabBarActiveTintColor: "#6851D6",
            tabBarInactiveTintColor: "#808080",
            headerShown: false,
          }}
        >
          <Tab.Screen
            name="Chats"
            component={ConversationsScreen}
            options={{
              tabBarIcon: ({ color, size }) => (
                <Icon name="chatbubbles-outline" size={size} color={color} />
              ),
            }}
          />
          <Tab.Screen
            name="Users"
            component={UsersScreen}
            options={{
              tabBarIcon: ({ color, size }) => (
                <Icon name="people-outline" size={size} color={color} />
              ),
            }}
          />
          <Tab.Screen
            name="Groups"
            component={GroupsScreen}
            options={{
              tabBarIcon: ({ color, size }) => (
                <Icon name="people-circle-outline" size={size} color={color} />
              ),
            }}
          />
          <Tab.Screen
            name="Calls"
            component={CallLogsScreen}
            options={{
              tabBarIcon: ({ color, size }) => (
                <Icon name="call-outline" size={size} color={color} />
              ),
            }}
          />
        </Tab.Navigator>
      );
    }

    // Main App Component
    export default function App() {
      return (
        <GestureHandlerRootView style={{ flex: 1 }}>
          <SafeAreaProvider>
            <CometChatI18nProvider>
              <CometChatThemeProvider>
                <NavigationContainer>
                  <Stack.Navigator screenOptions={{ headerShown: false }}>
                    <Stack.Screen name="Home" component={TabNavigator} />
                    <Stack.Screen name="Messages" component={MessagesScreen} />
                  </Stack.Navigator>
                </NavigationContainer>
              </CometChatThemeProvider>
            </CometChatI18nProvider>
          </SafeAreaProvider>
        </GestureHandlerRootView>
      );
    }
    ```
    </Tab>

## Customizing Tab Bar Appearance

You can customize the tab bar appearance using React Navigation's `screenOptions`:

<Tabs>
<Tab title="TypeScript">
```tsx
<Tab.Navigator
  screenOptions={{
    tabBarActiveTintColor: "#6851D6",
    tabBarInactiveTintColor: "#808080",
    tabBarStyle: {
      backgroundColor: "#FFFFFF",
      borderTopWidth: 1,
      borderTopColor: "#E8E8E8",
      paddingBottom: 5,
      paddingTop: 5,
      height: 60,
    },
    tabBarLabelStyle: {
      fontSize: 12,
      fontWeight: "500",
    },
    headerShown: false,
  }}
>
  {/* Tab screens */}
</Tab.Navigator>
```
</Tab>
