The `CometChatMessagePreview` component displays a preview of a message that is being replied to or edited. It provides a compact view with customizable title and subtitle templates, and handles deleted messages gracefully with a special UI.

## Overview

The MessagePreview component is designed for contextual message interactions:

- **Template Support**: Fully customizable title and subtitle via Angular TemplateRef
- **Deleted Message Handling**: Automatically displays special UI for deleted messages
- **Flexible Display Modes**: Supports both composer view (with close button) and bubble view (embedded)
- **Accessibility**: Full keyboard support for close button interaction
- **Responsive**: Adapts width based on parent container in bubble view mode

## Basic Usage

### Simple Message Preview

```html expandable
<cometchat-message-preview
  [message]="replyMessage"
  (closeClick)="onClosePreview()">
  <ng-template #previewTitle>
    <span>Replying to {{ replyMessage.sender.name }}</span>
  </ng-template>
  <ng-template #previewSubtitle>
    <span>{{ replyMessage.text }}</span>
  </ng-template>
</cometchat-message-preview>
```

```typescript expandable
      </div>
    </ng-template>

      align-items: center;
      gap: 8px;
      font-weight: 600;
    }
    .reply-icon {
      font-size: 16px;
    }
    .preview-subtitle {
      color: #666;
      overflow: hidden;
      text-overflow: ellipsis;
      white-space: nowrap;
    }
  `]

export class ReplyComposerComponent {
  replyMessage: CometChat.BaseMessage | null = null;

  clearReply(): void {
    this.replyMessage = null;
  }

    if (!message) return '';
    
    if (message.getType() === 'text') {
      return (message as CometChat.TextMessage).getText();
    } else if (message.getType() === 'image') {
      return '📷 Image';
    } else if (message.getType() === 'video') {
      return '🎥 Video';
    }
    return 'Message';
  }
}
```

```typescript expandable
import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { CometChat } from '@cometchat/chat-sdk-javascript';
import { CometChatMessagePreviewComponent } from '@cometchat/chat-uikit-angular';

@Component({
  selector: 'app-message-composer-with-reply',
  standalone: true,
  imports: [CommonModule, CometChatMessagePreviewComponent],
  template: `
    <div class="composer-container">
      <!-- Message Preview (shown when replying) -->
      @if (replyToMessage) {
        <cometchat-message-preview
          [previewTitle]="titleTemplate"
          [previewSubtitle]="subtitleTemplate"
          [message]="replyToMessage"
          (closeClick)="cancelReply()">
        </cometchat-message-preview>
      }

      <!-- Message Input -->
      <div class="message-input">
        <textarea
          [(ngModel)]="messageText"
          placeholder="Type a message..."
          (keydown.enter)="sendMessage($event)">
        </textarea>
        <button (click)="sendMessage()">Send</button>
      </div>
    </div>

    <!-- Templates -->
    <ng-template #titleTemplate>
      <div class="reply-title">
        <span class="reply-label">Replying to</span>
        <span class="sender-name">{{ replyToMessage?.sender?.name }}</span>
      </div>
    </ng-template>

    <ng-template #subtitleTemplate>
      <div class="reply-content">
        {{ getMessageContent(replyToMessage) }}
      </div>
    </ng-template>
  `,
  styles: [`
    .composer-container {
      display: flex;
      flex-direction: column;
      gap: 8px;
      padding: 12px;
      background: #f5f5f5;
      border-radius: 8px;
    }
    .message-input {
      display: flex;
      gap: 8px;
    }
    textarea {
      flex: 1;
      padding: 8px;
      border: 1px solid #ccc;
      border-radius: 4px;
      resize: none;
      min-height: 40px;
    }
    .reply-title {
      display: flex;
      flex-direction: column;
      gap: 2px;
    }
    .reply-label {
      font-size: 12px;
      color: #666;
    }
    .sender-name {
      font-weight: 600;
      color: #333;
    }
    .reply-content {
      font-size: 14px;
      color: #666;
      overflow: hidden;
      text-overflow: ellipsis;
      white-space: nowrap;
    }
  `]
})
export class MessageComposerWithReplyComponent {
  replyToMessage: CometChat.BaseMessage | null = null;
  messageText = '';

  setReplyMessage(message: CometChat.BaseMessage): void {
    this.replyToMessage = message;
  }

  cancelReply(): void {
    this.replyToMessage = null;
  }

  getMessageContent(message: CometChat.BaseMessage | null): string {
    if (!message) return '';
    
    const type = message.getType();
    if (type === 'text') {
      return (message as CometChat.TextMessage).getText();
    } else if (type === 'image') {
      return '📷 Photo';
    } else if (type === 'video') {
      return '🎥 Video';
    } else if (type === 'audio') {
      return '🎵 Audio';
    } else if (type === 'file') {
      return '📎 File';
    }
    return 'Message';
  }

  sendMessage(event?: KeyboardEvent): void {
    if (event && !event.shiftKey) {
      event.preventDefault();
    } else if (event) {
      return; // Allow Shift+Enter for new line
    }

    if (!this.messageText.trim()) return;

    // Send message with reply metadata
    console.log('Sending message:', this.messageText);
    console.log('Reply to:', this.replyToMessage?.getId());

    // Clear after sending
    this.messageText = '';
    this.replyToMessage = null;
  }
}
```

## Behavior

### Deleted Message Handling

When a message has been deleted (detected via `message.getDeletedAt()`), the component automatically displays a special UI:

- Shows a deleted message icon
- Displays localized "message_deleted" text
- Hides the subtitle template
- Maintains the same layout structure

```typescript
// The component automatically detects deleted messages
<cometchat-message-preview
  [message]="deletedMessage"
  [previewTitle]="titleTemplate"
  [previewSubtitle]="subtitleTemplate">
</cometchat-message-preview>

// Will show: [Icon] "This message was deleted"
```

### View Modes

The component supports two display modes:

**Composer View** (default):
- Shows close button
- Full width container
- Used in message composer for reply/edit

**Bubble View** (`hideCloseButton="true"`):
- No close button
- Responsive width based on parent
- Used inside message bubbles for quoted messages
- Automatically adjusts width using ResizeObserver

### Width Calculation

In bubble view mode, the component:
1. Observes the parent message bubble's content width
2. Adjusts its own width to match
3. Handles moderated messages with minimum 240px width
4. Updates dynamically on resize

## Accessibility

The MessagePreview component includes comprehensive accessibility features to ensure WCAG 2.1 Level AA compliance:

### Keyboard Support

The component is fully keyboard accessible with the following keyboard shortcuts:

| Key | Action | Context |
|-----|--------|---------|
| `Tab` | Moves focus to the close button | When close button is visible |
| `Shift + Tab` | Moves focus to previous focusable element | Reverse tab order |
| `Enter` | Activates the close button and closes the preview | When close button is focused |
| `Space` | Activates the close button and closes the preview | When close button is focused |

**Note**: The Space key press prevents default scrolling behavior to avoid unintended page scrolling.

### ARIA Attributes

The component implements proper ARIA attributes for screen reader support:

| Element | ARIA Attribute | Purpose |
|---------|---------------|---------|
| Container | `role="region"` | Identifies the preview as a distinct region |
| Container | `aria-label` | Provides accessible name for the region (customizable via `ariaLabel` input) |
| Close button | `role="button"` | Identifies the element as a button |
| Close button | `aria-label` | Provides localized "Close" label for screen readers |
| Close button | `tabindex="0"` | Makes the button keyboard focusable |
| Deleted message icon | `aria-hidden="true"` | Hides decorative icon from screen readers |

### Custom ARIA Labels

You can provide custom ARIA labels for better context:

```html
<cometchat-message-preview

  [ariaLabel]="'Reply to message from ' + replyMessage.sender.name"
```

### Focus Management

The component provides clear visual focus indicators:

- **Visible Focus Ring**: 2px solid outline in primary color when focused via keyboard
- **Focus Offset**: 2px offset for better visibility
- **Focus-Visible Support**: Focus indicators only appear for keyboard navigation, not mouse clicks
- **WCAG Contrast**: Focus indicators meet WCAG 2.1 Level AA contrast requirements (3:1 minimum)

```css
/* Focus indicator styling */
.cometchat-message-preview__close:focus-visible {

  outline-offset: 2px;

}
```

### Nested Interactive Elements

The component supports nested interactive elements within templates:

**Requirements**:
- All nested interactive elements (buttons, links) in `previewTitle` and `previewSubtitle` templates are independently focusable
- Tab order follows visual layout: title → subtitle → close button
- Nested element activation does not trigger parent actions
- Each nested element maintains its own focus state

**Example with nested elements**:

```html expandable
<cometchat-message-preview
  [previewTitle]="titleWithButton"
  [previewSubtitle]="subtitleWithLink"
  [message]="message"
  (closeClick)="onClose()">

  <div class="title-container">
    <span>Replying to {{ message.sender.name }}</span>
    <button (click)="viewProfile($event)" aria-label="View profile">

    </button>
  </div>
</ng-template>

<ng-template #subtitleWithLink>
  <div class="subtitle-container">
    <a [href]="messageLink" target="_blank" rel="noopener">
      View original message
    </a>
  </div>
</ng-template>
```

**Best Practices for Nested Elements**:
1. Ensure all interactive elements have proper `aria-label` attributes
2. Use semantic HTML (`<button>`, `<a>`) instead of `<div>` with click handlers
3. Stop event propagation in nested element handlers to prevent parent activation
4. Maintain logical tab order by placing elements in visual order in the DOM

### Keyboard Accessibility Implementation

The component implements keyboard accessibility through event handlers:

```typescript
// Keyboard event handler for close button
onCloseKeyDown(event: KeyboardEvent): void {
  if (event.key === 'Enter' || event.key === ' ') {
    event.preventDefault(); // Prevent default space scrolling
    this.handleClose();
  }
}
```

This ensures:
- Enter and Space keys activate the close button
- Space key doesn't cause page scrolling
- Other keys are ignored (no unintended actions)
- Consistent behavior with native button elements

## Best Practices

<Tip>
  Always provide both title and subtitle templates for a complete preview experience. The title typically shows who sent the message, while the subtitle shows the content.
</Tip>

<Warning>
  The component automatically handles deleted messages. Don't try to hide the component when a message is deleted - let it show the deleted message UI instead.
</Warning>

<Info>
  Use `hideCloseButton="true"` when embedding the preview inside message bubbles. Use the default mode (with close button) in message composers.
</Info>

<Tip>
  Keep subtitle content concise. Use text truncation (ellipsis) for long messages to maintain a clean preview.
</Tip>
