Add the CometChat Widget by pasting one small HTML snippet. It’s a plug-and-play chat window—just decide how people should enter the chat and follow the matching option below:

<Steps className="mt-6">
  <Step title="Guest / Anonymous Users">

  <Step title="Create and Login User on the Fly">

## 1. Guest / Anonymous Users (No Login)

**Use this when:**

- Anyone should be able to chat right away  
- You do not have (or do not want) your own login form yet  

### 1.1 Code for `<head>`

```html
<script defer src="https://cdn.jsdelivr.net/npm/@cometchat/chat-embed@1.x.x/dist/main.js"></script>
```

### 1.2 Code for `<body>`

```html lines highlight={9-10, 12, 16-18, 23-25}
<!-- Where the chat widget should appear -->
<div id="cometChatMount"></div>

<script type="module">
  import { CometChatAuth } from "https://cdn.example.com/cometchat/cometchat-auth.js";

  //  EDIT ONLY THE TEXT INSIDE QUOTES
  const COMETCHAT_WIDGET_CONFIG = {
    appID: "YOUR_APP_ID",
    region: "YOUR_APP_REGION",   // e.g. "us", "eu", "in"
    mode: "guest",               // keep this as "guest"
    authKey: "YOUR_AUTH_KEY",

    // Optional: how the guest appears in chat
    user: {
      name: "Guest User",
      avatar: "",                // e.g. "https://example.com/guest.png"
      link: ""                   // e.g. "https://example.com/profile"
    },

    // Widget placement + size
    mount: "#cometChatMount",
    width: "450px",
    height: "80vh",
    isDocked: true               // true = docked widget (recommended)
  };

  CometChatAuth.start(COMETCHAT_WIDGET_CONFIG);
</script>
```

## 2. Create User on the Fly (UID Login)

**Use this when:**

- You already give each person a unique ID (email, username, etc.)  
- You want them to look the same every time they chat  

### 2.1 Code for `<head>`

```html
<script defer src="https://cdn.jsdelivr.net/npm/@cometchat/chat-embed@1.x.x/dist/main.js"></script>
```

### 2.2 Code for `<body>`

```html lines highlight={9-10, 12, 15, 19-21, 26-28}
<!-- Where the chat widget should appear -->
<div id="cometChatMount"></div>

<script type="module">
  import { CometChatAuth } from "https://cdn.example.com/cometchat/cometchat-auth.js";

  //  EDIT ONLY THE TEXT INSIDE QUOTES
  const COMETCHAT_WIDGET_CONFIG = {
    appID: "YOUR_APP_ID",
    region: "YOUR_APP_REGION",      // e.g. "us", "eu", "in"
    mode: "uid",                    // keep this as "uid"
    authKey: "YOUR_AUTH_KEY",

    // IMPORTANT: this must be unique per user
    uid: "UNIQUE_USER_ID",          // e.g. "user_123", "customer_42"

    // Optional: user profile shown in chat
    user: {
      name: "User Name",            // e.g. "Alex Johnson"
      avatar: "https://example.com/uid.png",
      link: "https://example.com/profile/uid"
    },

    // Widget placement + size
    mount: "#cometChatMount",
    width: "450px",
    height: "80vh",
    isDocked: true
  };

  CometChatAuth.start(COMETCHAT_WIDGET_CONFIG);
</script>
```

## 3. App With Its Own Backend (Auth Token Login)

**Use this when:**

- People log in through your own backend  
- Your backend can request CometChat auth tokens after login  

### 3.1 Code for `<head>`

```html
<script defer src="https://cdn.jsdelivr.net/npm/@cometchat/chat-embed@1.x.x/dist/main.js"></script>
```

### 3.2 Code for `<body>`

```html lines highlight={9-10, 14, 18-19}
<!-- Where the chat widget should appear -->
<div id="cometChatMount"></div>

<script type="module">
  import { CometChatAuth } from "https://cdn.example.com/cometchat/cometchat-auth.js";

  //  EDIT ONLY THE TEXT INSIDE QUOTES
  const COMETCHAT_WIDGET_CONFIG = {
    appID: "YOUR_APP_ID",
    region: "YOUR_APP_REGION",       // e.g. "us", "eu", "in"
    mode: "authToken",               // keep this as "authToken"

    // Generated BY YOUR BACKEND after user login
    authToken: "USER_AUTH_TOKEN",

    // Widget placement + size
    mount: "#cometChatMount",
    width: "450px",
    height: "80vh",
    isDocked: true
  };

  CometChatAuth.start(COMETCHAT_WIDGET_CONFIG);
</script>
```
